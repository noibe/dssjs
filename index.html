DSS.JS Introduction

Dynamic Style Sheets

Using

To use the DSS.JS, download the dss.js and put the stylesheet object array to addStyle() function:

addStyle(styleObject, true);

Rule Model (object)

{
	selector: 'body',
	properties: [
		{
			important: true,
			name: 'color',
			value: 'green'
		},
		{
			name: 'color',
			value: 'blue'
		}
	]
}

Rule Model (array alternative)

[
	'body',
	[
		[
			'background-color',
			'red',
			true
		],
		[
			'color',
			'blue'
		]
	]
]
